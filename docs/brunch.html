exports.path = (paths...) ->
  escapedFiles = paths.map((pattern) ->
    pattern.replace(/[\-\[\]\/\{\}\(\)\+\?\.\\\^\$\|]/g, "\\$&")
  ).join('|')
  new RegExp('^' + escapedFiles)
